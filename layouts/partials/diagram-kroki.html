{{- /* Cache-first Kroki diagram: checks assets/kroki-cache/.svg, falls back to API. Cache is pre-generated by scripts/cache-diagrams.py. */ -}} {{- $type := .type -}} {{- $source := .source -}} {{- $cacheKey := printf "%s\n%s\n%s" $type $source (default (dict) .options | jsonify) -}} {{- $cacheHash := crypto.SHA256 $cacheKey -}} {{- $svg := "" -}} {{- $cachePath := printf "kroki-cache/%s.svg" $cacheHash -}} {{- with resources.Get $cachePath -}} {{- $svg = .Content -}} {{- end -}} {{- if not $svg -}} {{- $url := printf "https://kroki.io/%s/svg" $type -}} {{- $body := dict "diagram_source" $source "output_format" "svg" -}} {{- with .options -}} {{- $body = merge $body (dict "diagram_options" .) -}} {{- end -}} {{- $opts := dict "method" "post" "body" ($body | jsonify) "headers" (dict "Content-Type" "application/json") -}} {{- with resources.GetRemote $url $opts -}} {{- $svg = .Content -}} {{- end -}} {{- end -}} {{- if $svg -}} {{- else -}} {{- errorf "Kroki diagram failed for %s (no cache, API unavailable)" $type -}} {{- end -}}
